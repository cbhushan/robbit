00293 void Frame_Display::reshape(int w, int h) {
00294   //prevent a divide by zero, when window is too short
00295   if(h == 0) { h = 1; }
00296   aspect = 1.0 * w / h;
00297 
00298   // reset the coordinate system before modifying
00299   glMatrixMode(GL_PROJECTION);
00300   glLoadIdentity();     // no transformations
00301 
00302   // set the viewport to be the entire window
00303   glViewport(0, 0, w, h);
00304 
00305   upX = upY = 0;
00306   upZ = 1.0f;
00307         
00308   // set the view.
00309   gluPerspective(45,aspect,this->Near,this->Far);
00310 
00311   // Sets the camera to look at the centroid of the robots
00312   if(!camera_on_bot) {
00313 
00314     // If top-view, set the camera vertically above center
00315     if(topview) {
00316       eyeX = view_centerX;
00317       eyeY = view_centerY;
00318       eyeZ = radius_camera_movement;
00319 
00320       upX = upZ = 0;
00321       upY = 1.0f;
00322     }
00323     else {
00324       // general view
00325           if(this->autoview) {
00326                 AutoUpdateEye();
00327       }
00328       eyeX = view_centerX + (radius_camera_movement*sin(theta*radianFactor) * cos(phi*radianFactor));
00329       eyeY = view_centerY + (radius_camera_movement*sin(theta*radianFactor) * sin(phi*radianFactor));
00330       eyeZ = radius_camera_movement * cos(radianFactor*theta);
00331     }
00332   }
00333   //sets the onboard view of a robot, Sets the camera over specified robot
00334   else {
00335     eyeX = data.bot_x[camera_on_bot-1];
00336     eyeY = data.bot_y[camera_on_bot-1];
00337     eyeZ = height_robot * 2;
00338                 
00339     view_centerX = eyeX + (radius_of_robot) * cos(data.bot_orient[camera_on_bot-1]);
00340     view_centerY = eyeY + (radius_of_robot) * sin(data.bot_orient[camera_on_bot-1]);
00341     view_centerZ = eyeZ;
00342   }
00343 
00344   gluLookAt(eyeX, eyeY, eyeZ, view_centerX, view_centerY, view_centerZ, upX, upY, upZ); 
00345   glMatrixMode(GL_MODELVIEW);
00346   glLoadIdentity();
00347 }
