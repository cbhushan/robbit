00297 void Frame_Display::reshape(int w, int h) {
00298   //prevent a divide by zero, when window is too short
00299   if(h == 0) { h = 1; }
00300   aspect = 1.0 * w / h;
00301 
00302   // reset the coordinate system before modifying
00303   glMatrixMode(GL_PROJECTION);
00304   glLoadIdentity();     // no transformations
00305 
00306   // set the viewport to be the entire window
00307   glViewport(0, 0, w, h);
00308 
00309   upX = upY = 0;
00310   upZ = 1.0f;
00311         
00312   // set the view.
00313   gluPerspective(45,aspect,this->Near,this->Far);
00314 
00315   // Sets the camera to look at the centroid of the robots
00316   if(!camera_on_bot) {
00317 
00318     // If top-view, set the camera vertically above center
00319     if(topview) {
00320       eyeX = view_centerX;
00321       eyeY = view_centerY;
00322       eyeZ = radius_camera_movement;
00323 
00324       upX = upZ = 0;
00325       upY = 1.0f;
00326     }
00327     else {
00328       // general view
00329       if(this->autoview) {
00330         AutoUpdateEye();
00331       }
00332 
00333       eyeX = view_centerX + (radius_camera_movement*sin(theta*radianFactor) * cos(phi*radianFactor));
00334       eyeY = view_centerY + (radius_camera_movement*sin(theta*radianFactor) * sin(phi*radianFactor));
00335       eyeZ = radius_camera_movement * cos(radianFactor*theta);
00336     }
00337   }
00338   //sets the onboard view of a robot, Sets the camera over specified robot
00339   else {
00340     eyeX = data.bot_x[camera_on_bot-1];
00341     eyeY = data.bot_y[camera_on_bot-1];
00342     eyeZ = height_robot * 2;
00343                 
00344     view_centerX = eyeX + (radius_of_robot) * cos(data.bot_orient[camera_on_bot-1]);
00345     view_centerY = eyeY + (radius_of_robot) * sin(data.bot_orient[camera_on_bot-1]);
00346     view_centerZ = eyeZ;
00347   }
00348 
00349   gluLookAt(eyeX, eyeY, eyeZ, view_centerX, view_centerY, view_centerZ, upX, upY, upZ); 
00350   glMatrixMode(GL_MODELVIEW);
00351   glLoadIdentity();
00352 }

00376     fl_message("Cannot open file %s !",fname);

00380 
00381     // reads a line from the file
00382     obstacle_data_file.getline(string, 1000);

00386       obstacle_data_file.getline(string, 1000);

00390       if(string[0]!='%') {
00391         // Copies first two letters to identify type of the obstacle
